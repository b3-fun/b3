## ParentProvider

El componente `ParentProvider` conecta tu frontend de juego con la plataforma Upside y proporciona autenticaci√≥n.

```javascript

export default function CoinFlipGame() {
  return (
    <ParentProvider>
      <GameContent />
    </ParentProvider>
  );
}

function GameContent() {
  const { token, balance, showWinModal, showLossModal, refetchBalance } = useParent();

  // O usa hooks individuales:
  const balance = useBalance(); // number | null
  const token = useToken(); // string | null
  const authFetch = useAuthenticatedFetch(); // fetch con encabezado Authorization

  return (
    <div>
      <h1>Coin Flip</h1>
      <p>Balance: {(balance / 1e18).toFixed(2)} WIN</p>
      <button onClick={() => playGame(token)}>Jugar</button>
    </div>
  );
}
```

## Hooks Disponibles

### useParent()

Accede al contexto completo (token, balance, showWinModal, showLossModal, etc.)

**Valores de Retorno:**

- `token` (string | null): Token de autenticaci√≥n JWT para llamadas al backend
- `balance` (number | null): Saldo actual de tokens WIN en wei
- `showWinModal(wins: string)`: Muestra el modal de victoria
- `showLossModal(loss: string)`: Muestra el modal de derrota
- `showToast(options)`: Muestra una notificaci√≥n tipo toast
- `showCustomModal(content)`: Muestra un modal personalizado
- `refetchBalance()`: Solicita la actualizaci√≥n del saldo desde la plataforma

## Realizando Llamadas a la API

Usa el hook `useAuthenticatedFetch()` para hacer solicitudes autenticadas:

```javascript
import { useAuthenticatedFetch, useBalance, useToken } from "@b3dotfun/upside-sdk";

function GameComponent() {
  const authFetch = useAuthenticatedFetch();
  const balance = useBalance();
  const token = useToken();

  const playGame = async prediction => {
    // El encabezado Authorization se a√±ade autom√°ticamente
    const response = await authFetch("/api/game/coin-flip", {
      method: "POST",
      body: JSON.stringify({
        playerId: "player-id",
        prediction,
        betAmount: "1000000000000000000", // 1 WIN en wei
      }),
    });

    const data = await response.json();
    return data;
  };

  return (
    <div>
      <p>Balance: {(balance / 1e18).toFixed(2)} WIN</p>
      <button onClick={() => playGame("heads")}>Predecir Cara</button>
    </div>
  );
}
```

**C√≥mo Funciona useAuthenticatedFetch():**

- Incluye autom√°ticamente el encabezado `Authorization: Bearer {token}`
- Establece `Content-Type: application/json` por defecto
- Fusiona encabezados adicionales que proporciones
- Maneja el token de `useToken()` autom√°ticamente

## Ejemplo de Frontend

```javascript
import { ParentProvider, useParentContext } from "@b3dotfun/upside-sdk";

export default function CoinFlipGame() {
  return (
    <ParentProvider>
      <CoinFlipContent />
    </ParentProvider>
  );
}

function CoinFlipContent() {
  const { token, balance, playerId } = useParentContext();
  const [gameState, setGameState] = useState("ready"); // ready, playing, won, lost
  const [prediction, setPrediction] = useState(null);
  const [result, setResult] = useState(null);
  const [earnings, setEarnings] = useState(0);

  const playGame = async playerPrediction => {
    setPrediction(playerPrediction);
    setGameState("playing");

    try {
      // Llama a tu backend
      const response = await fetch("/api/game/coin-flip", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          Authorization: `Bearer ${token}`,
        },
        body: JSON.stringify({
          playerId,
          prediction: playerPrediction,
          betAmount: "100000000000000000", // 1 token en wei
        }),
      });

      const data = await response.json();

      if (data.outcome === "win") {
        setGameState("won");
        setEarnings(data.payout);
      } else {
        setGameState("lost");
        setEarnings(0);
      }

      setResult(data.result);
    } catch (error) {
      console.error("Error en el juego:", error);
      setGameState("error");
    }
  };

  return (
    <div style={{ textAlign: "center", padding: "40px" }}>
      <h1>Coin Flip</h1>
      <p>Balance: {(balance / 1e18).toFixed(2)} WIN</p>

      {gameState === "ready" && (
        <div>
          <button onClick={() => playGame("heads")}>Predecir Cara</button>
          <button onClick={() => playGame("tails")}>Predecir Cruz</button>
        </div>
      )}

      {gameState === "playing" && <p>Volteando...</p>}

      {gameState === "won" && (
        <div>
          <p>üéâ ¬°Has ganado! La moneda cay√≥ en {result}</p>
          <p>+{(earnings / 1e18).toFixed(2)} WIN</p>
        </div>
      )}

      {gameState === "lost" && (
        <div>
          <p>‚ùå ¬°Has perdido! La moneda cay√≥ en {result}</p>
          <p>Mejor suerte la pr√≥xima vez</p>
        </div>
      )}

      {gameState !== "ready" && <button onClick={() => setGameState("ready")}>Jugar de Nuevo</button>}
    </div>
  );
}
```
