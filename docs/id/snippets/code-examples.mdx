# Contoh Kode

## Instalasi

```bash title="Install B3 SDK" icon="download"
npm install @b3dotfun/sdk
```

## Pengaturan Dasar

```tsx title="Import Komponen SDK" icon="react"
import "@b3dotfun/sdk/index.css";
import { AnyspendProvider, B3Provider } dari "@b3dotfun/sdk";
```

## Komponen Pembelian NFT

```tsx title="Contoh Tombol NFT" icon="shopping-cart" lines
import { AnySpendNFTButton } dari "@b3dotfun/sdk/anyspend/react";

function NFTCard() {
  return (
    <AnySpendNFTButton
      nftContract={{
        chainId: 8333,
        contractAddress: "0x9c275ff1634519E9B5449ec79cd939B5F900564d",
        price: "500000000000000000",
        currency: { symbol: "ETH", decimals: 18 },
      }}
      recipientAddress={userAddress}
      onSuccess={txHash => showSuccess(txHash)}
    />
  );
}
```

## Otentikasi

```tsx title="Pengaturan Otentikasi B3" icon="user-shield" expandable
import { B3Provider, SignInWithB3 } dari "@b3dotfun/sdk/global-account/react";

function App() {
  return (
    <B3Provider environment="production">
      <SignInWithB3
        provider={{ strategy: "google" }}
        partnerId="your-partner-id"
        onLoginSuccess={globalAccount => {
          console.log("Authenticated:", globalAccount);
        }}
      />
    </B3Provider>
  );
}
```

## Koneksi Dompet

```tsx title="Hubungkan Dompet ke B3" icon="wallet" focus={8-12}
import { useConnect } dari "@b3dotfun/sdk";

function WalletButton() {
  const { connect, isConnecting } = useConnect();

  const handleConnect = async () => {
    try {
      await connect({
        chainId: 8333,
        rpcUrl: "https://mainnet-rpc.b3.fun",
      });
    } catch (error) {
      console.error("Connection failed:", error);
    }
  };

  return (
    <button onClick={handleConnect} disabled={isConnecting}>
      {isConnecting ? "Connecting..." : "Connect to B3"}
    </button>
  );
}
```
