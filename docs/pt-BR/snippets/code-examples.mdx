# Exemplos de Código

## Instalação

```bash title="Instalar B3 SDK" icon="download"
npm install @b3dotfun/sdk
```

## Configuração Básica

```tsx title="Importar Componentes do SDK" icon="react"
import "@b3dotfun/sdk/index.css";
import { AnyspendProvider, B3Provider } from "@b3dotfun/sdk";
```

## Componente de Compra de NFT

```tsx title="Exemplo de Botão NFT" icon="shopping-cart" lines
import { AnySpendNFTButton } from "@b3dotfun/sdk/anyspend/react";

function NFTCard() {
  return (
    <AnySpendNFTButton
      nftContract={{
        chainId: 8333,
        contractAddress: "0x9c275ff1634519E9B5449ec79cd939B5F900564d",
        price: "500000000000000000",
        currency: { symbol: "ETH", decimals: 18 },
      }}
      recipientAddress={userAddress}
      onSuccess={txHash => showSuccess(txHash)}
    />
  );
}
```

## Autenticação

```tsx title="Configuração de Autenticação B3" icon="user-shield" expandable
import { B3Provider, SignInWithB3 } from "@b3dotfun/sdk/global-account/react";

function App() {
  return (
    <B3Provider environment="production">
      <SignInWithB3
        provider={{ strategy: "google" }}
        partnerId="your-partner-id"
        onLoginSuccess={globalAccount => {
          console.log("Autenticado:", globalAccount);
        }}
      />
    </B3Provider>
  );
}
```

## Conexão com a Carteira

```tsx title="Conectar Carteira ao B3" icon="wallet" focus={8-12}
import { useConnect } from "@b3dotfun/sdk";

function WalletButton() {
  const { connect, isConnecting } = useConnect();

  const handleConnect = async () => {
    try {
      await connect({
        chainId: 8333,
        rpcUrl: "https://mainnet-rpc.b3.fun",
      });
    } catch (error) {
      console.error("Conexão falhou:", error);
    }
  };

  return (
    <button onClick={handleConnect} disabled={isConnecting}>
      {isConnecting ? "Conectando..." : "Conectar ao B3"}
    </button>
  );
}
```
